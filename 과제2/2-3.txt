public class Solution {
    public static int solution(String[] order) {
        int total = 0; // 총 금액 초기화
        for (String item : order) {
            // "anything" 또는 "americano"는 차가운 아메리카노로 처리
            if (item.equals("anything") || item.contains("americano")) {
                total += 4500; // 아메리카노 가격
            } 
            // "cafelatte" 또는 카페 라테를 포함한 경우 처리
            else if (item.contains("cafelatte")) {
                total += 5000; // 카페 라테 가격
            }
        }
        return total; // 총 결제 금액 반환
    }

    public static void main(String[] args) {
        // 테스트 예제
        System.out.println(solution(new String[]{"cafelatte", "americanoice", "hotcafelatte", "anything"})); // 출력: 19000
        System.out.println(solution(new String[]{"americanoice", "americano", "iceamericano"})); // 출력: 13500
    }
}
