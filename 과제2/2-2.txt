public class Solution {
    public static int solution(String my_string, String is_prefix) {
        // 접두사 확인
        if (my_string.startsWith(is_prefix)) {
            return 1; // 접두사일 경우
        } else {
            return 0; // 접두사가 아닐 경우
        }
    }

    public static void main(String[] args) {
        // 테스트 예제
        System.out.println(solution("banana", "ban")); // 출력: 1
        System.out.println(solution("banana", "nan")); // 출력: 0
        System.out.println(solution("banana", "abcd")); // 출력: 0
        System.out.println(solution("banana", "bananan")); // 출력: 0
    }
}
